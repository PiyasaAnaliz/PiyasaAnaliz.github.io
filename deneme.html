import requests import time # Buraya kendi API anahtarınızı yapıştırın API_KEY = "YOUR_API_KEY" def get_gold_prices(): url = "https://api.collectapi.com/economy/goldPrice" headers = { 'content-type': 'application/json', 'authorization': f'apikey {API_KEY}' } try: response = requests.get(url, headers=headers) data = response.json() if data['success']: print("-" * 30) for item in data['result']: # Gram Altın, Çeyrek Altın vb. verileri yazdırıyoruz print(f"{item['name']}: {item['buying']} TL - {item['selling']} TL") else: print("Veri çekilemedi.") except Exception as e: print(f"Bir hata oluştu: {e}") # Döngü ile anlık çalıştırma while True: get_gold_prices() # 10 saniyede bir güncelle time.sleep(10)
