import requests import time import os # Türkiye piyasası verisi sağlayan ücretsiz uç nokta URL = "https://api.doviz.com/api/v1/currencies/gram-altin/latest" def fetch_gold_price(): try: response = requests.get(URL, timeout=5) if response.status_code == 200: data = response.json() # Veri yapısını alıyoruz buying = data['buying'] selling = data['selling'] update_date = data['updateDate'] # Ekranı temizle (daha temiz bir simülasyon görüntüsü için) os.system('cls' if os.name == 'nt' else 'clear') print("=== GERÇEK ZAMANLI ALTIN TAKİP ===") print(f"Gram Altın (Alış): {buying} TL") print(f"Gram Altın (Satış): {selling} TL") print(f"Son Güncelleme: {update_date}") print("==================================") else: print("Veri kaynağına ulaşılamadı.") except Exception as e: print(f"Bağlantı hatası: {e}") # Döngü print("Sistem başlatılıyor...") while True: fetch_gold_price() time.sleep(5) # 5 saniyede bir güncelle
